Service Integration Pattern
============================================================

OLD CODE (Without Activity Tracking):
--------------------------------------
@app.route('/api/service/action', methods=['POST'])
@require_auth
def service_action(current_user):
    data = request.get_json()
    result = service.process(data)
    return jsonify(result), 200


NEW CODE (With Activity Tracking):
-----------------------------------
@app.route('/api/service/action', methods=['POST'])
@require_auth
def service_action(current_user):
    import time
    from app.utils.activity_helper import log_service_action
    
    start_time = time.time()
    data = request.get_json()
    result = service.process(data)
    
    # âœ¨ ADD ACTIVITY LOGGING âœ¨
    duration_ms = (time.time() - start_time) * 1000
    log_service_action(
        user_id=current_user['user_id'],
        params=data,
        result=result,
        duration_ms=duration_ms
    )
    
    return jsonify(result), 200
